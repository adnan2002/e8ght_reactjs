To check if a user is a freelancer, make an authenticated GET request using the hooks to "/users/me/freelancer/", if the response is ok, then you can move on. but if it isn't make sure to redirect the user to FreelancerForm.jsx (a file you will create).
Make sure checking the endpoint is the last check meaning first check if the user is authenticated, then the user has user.role = freelancer and then you can check the endpoint. This is only for freelancer, ignore customer for now.

for the FreelancerForm.jsx, just create a very simple file that says freelancer form nothing fancy.

these checks should be in createWithRoleAuth.jsx




route: "/users/me/freelancer/"
request type: GET
method:
func (s *Server) getFreelancerByUserIDHandler(c *gin.Context) {
	payloadVal, ok := c.Get(authenticatedPayload)
	if !ok {
		c.JSON(http.StatusUnauthorized, gin.H{"error": "unauthorized"})
		return
	}
	payload := payloadVal.(token.Payload)
	userID := payload.UserID

	if payload.Role != "freelancer" {
		c.JSON(http.StatusForbidden, gin.H{"error": "unauthorized"})
		return
	}

	freelancer, err := s.Store.GetFreelancerByUserID(c, userID)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusOK, gin.H{"freelancer": freelancer})
}

response: {
	"freelancer": ...
}


make sure to handle the response status types: internal server, forbidden, unauthorized, etc.
