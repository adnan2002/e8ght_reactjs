from the freelancer dashbaord i want you to change the patch api endpoint from "/users/me/freelancer/" to


1- PATCH "/users/me/freelancer/is-accepting-orders" which toggles the is accepting order section of freelancer
2- PATCH "/users/me/freelancer/is-public" which toggles the is public section of freelancer


type setFreelancerIsAcceptingOrdersRequest struct {
	IsAcceptingOrders bool `json:"is_accepting_orders" binding:"required"`
}

func (s *Server) setFreelancerIsAcceptingOrdersHandler(c *gin.Context) {
	var req setFreelancerIsAcceptingOrdersRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	payloadVal, ok := c.Get(authenticatedPayload)
	if !ok {
		c.JSON(http.StatusUnauthorized, gin.H{"error": "unauthorized"})
		return
	}
	payload := payloadVal.(token.Payload)
	userID := payload.UserID
	if payload.Role != "freelancer" {
		c.JSON(http.StatusForbidden, gin.H{"error": "unauthorized"})
		return
	}
	freelancer, err := s.Store.SetFreelancerIsAcceptingOrders(c, db.SetFreelancerIsAcceptingOrdersParams{
		UserID:            userID,
		IsAcceptingOrders: req.IsAcceptingOrders,
	})
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	c.JSON(http.StatusOK, gin.H{"freelancer": freelancer})
}
